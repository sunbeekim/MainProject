CREATE TABLE board_invitations (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    board_id BIGINT NOT NULL,
    inviter_email VARCHAR(100) NOT NULL,
    invitee_email VARCHAR(100) NOT NULL,
    invitation_code VARCHAR(50) NOT NULL,
    status VARCHAR(20) DEFAULT 'PENDING',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expired_at TIMESTAMP,
    FOREIGN KEY (board_id) REFERENCES boards(id) ON DELETE CASCADE,
    FOREIGN KEY (inviter_email) REFERENCES users(email) ON DELETE CASCADE,
    FOREIGN KEY (invitee_email) REFERENCES users(email) ON DELETE CASCADE,
    UNIQUE KEY (invitation_code)
);
